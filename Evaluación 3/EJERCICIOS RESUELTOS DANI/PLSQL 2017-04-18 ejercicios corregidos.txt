DECLARE
	horas number (8,0):=0;
	minutos number (8,0):=0;
	segundos number (8,0):=0;
	tiempo number (8,0):=&t;
BEGIN
	minutos:=tiempo/60;
	segundos:=tiempo mod 60;
	horas:=minutos/60;
	minutos:=minutos mod 60;
	DBMS_OUTPUT.PUT_LINE('Hora:'||horas||':'||minutos||':'||segundos);
END;
