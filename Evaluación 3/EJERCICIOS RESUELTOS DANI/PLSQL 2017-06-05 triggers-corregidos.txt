	CREATE OR REPLACE TRIGGER HOJ1_2
	BEFORE UPDATE OF MSAL
	ON EMPLOYEES
	FOR EACH ROW
	BEGIN
		insert into empleados_audit
		values(:old.empno, :old.msal, :new.msal, sysdate);
	EXCEPTION
		WHEN OTHERS THEN
			DBMS_OUTPUT.PUT_LINE('Error inesperado');
	END;
